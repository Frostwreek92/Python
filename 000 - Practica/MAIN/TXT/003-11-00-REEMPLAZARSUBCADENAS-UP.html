# Reemplazar subcadenas

cadena = 'Hola, mundo!'
print(f'Cadena original: {cadena}')
nueva_cadena = cadena.replace('mundo', 'a todos')
print(f'Nueva cadena reemplazada: {nueva_cadena}' )

# Sustituir hola por adios
nueva_cadena = cadena.replace('Hola', 'Adios')
print(f'Nueva cadena reemplazada: {nueva_cadena}')
